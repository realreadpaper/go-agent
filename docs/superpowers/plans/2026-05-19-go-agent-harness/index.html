internal/tools/registry.go

package tools

import "learn-claude-code-go/internal/llm"

type Handler func(input map[string]any) (string, error)

type Tool struct {
	Spec    llm.ToolSpec
	Handler Handler
}

type Registry struct {
	tools map[string]Tool
}

func NewRegistry() *Registry {
	return &Registry{tools: map[string]Tool{}}
}

func (r *Registry) Register(t Tool) {
	r.tools[t.Spec.Name] = t
}

func (r *Registry) Specs() []llm.ToolSpec {
	specs := make([]llm.ToolSpec, 0, len(r.tools))
	for _, tool := range r.tools {
		specs = append(specs, tool.Spec)
	}
	return specs
}

func (r *Registry) Run(name string, input map[string]any) string {
	tool, ok := r.tools[name]
	if !ok {
		return "Error: unknown tool: " + name
	}
	out, err := tool.Handler(input)
	if err != nil {
		return "Error: " + err.Error()
	}
	if out == "" {
		return "(no output)"
	}
	return out
}

internal/agent/loop.go

package agent

import "learn-claude-code-go/internal/llm"

type BeforeCallHook func(messages *[]llm.Message) error
type AfterToolHook func(name string)

type Loop struct {
	Client     llm.Client
	Model      string
	System     string
	Tools      ToolRunner
	MaxTokens  int
	MaxRounds  int
	BeforeCall []BeforeCallHook
	AfterTool  []AfterToolHook
}

type ToolRunner interface {
	Specs() []llm.ToolSpec
	Run(name string, input map[string]any) string
}

func (l *Loop) Run(messages []llm.Message) ([]llm.Message, llm.Response, error) {
	var resp llm.Response
	rounds := l.MaxRounds
	if rounds == 0 {
		rounds = 50
	}
	for i := 0; i < rounds; i++ {
		for _, hook := range l.BeforeCall {
			if err := hook(&messages); err != nil {
				return messages, resp, err
			}
		}
		var err error
		resp, err = l.Client.Create(llm.Request{
			Model: l.Model, System: l.System, Messages: messages,
			Tools: l.Tools.Specs(), MaxTokens: l.MaxTokens,
		})
		if err != nil {
			return messages, resp, err
		}
		messages = append(messages, llm.Message{Role: "assistant", Content: resp.Content})
		if resp.StopReason != "tool_use" {
			return messages, resp, nil
		}
		results := make([]llm.ToolResult, 0)
		for _, block := range resp.Content {
			if block.Type != "tool_use" {
				continue
			}
			output := l.Tools.Run(block.Name, block.Input)
			for _, hook := range l.AfterTool {
				hook(block.Name)
			}
			results = append(results, llm.ToolResult{Type: "tool_result", ToolUseID: block.ID, Content: output})
		}
		messages = append(messages, llm.Message{Role: "user", Content: results})
	}
	return messages, resp, nil
}

给第一次接触 Agent 的读者

这一节先不急着实现 agent 行为，而是先做“模型连接层”。Agent 的每一步都要问一次模型：现在看到这些消息和工具，你下一步要说话还是调用工具？如果这层直接散落在业务代码里，后面每加一个机制都要重新处理 API 请求、响应解析、错误和测试。

新手先懂这几个概念

  • Message：对话历史中的一条记录。用户输入、助手回复、工具结果都会变成 message，agent loop 靠它们让模型知道刚才发生了什么。
  • ToolSpec：告诉模型“有哪些工具可以用、参数长什么样”。模型不会真的执行工具，它只会输出一个结构化的 tool call。
  • Response：模型返回的内容，可能是普通文本，也可能是 tool_use。后续循环会根据 stop_reason 决定是否执行工具。
  • Scripted test client：测试用的脚本化模型。业务包可以在自己的测试里实现最小 fake，不需要把 fake provider 放进生产 internal/llm。

为什么这样拆

internal/llm 只负责协议转换，不知道 agent、任务、文件系统或团队协作。这样后续从 OpenAI Responses 换成 OpenAI-compatible Chat 或 DeepSeek adapter 时，agent 主流程不需要改。第一次读代码时，可以把这层看成“模型插座”：外面接不同 provider，里面统一输出 Go 结构体。

完成后应该理解

你应该能解释一次 LLM 调用需要哪些输入、模型如何表达“我要调用工具”、为什么业务测试要避免真实网络，以及为什么 provider adapter 不能执行工具。

Step 3: 实现 OpenAI Responses adapter**

要求：

  • 从 OPENAI_API_KEY 读取 key。
  • OPENAI_BASE_URL 可覆盖，默认 https://api.openai.com。
  • 请求 POST /v1/responses。
  • ToolSpec 转为 OpenAI function tool。
  • ContentBlock{Type:"tool_use"} 转为 Responses API 续轮输入。
  • 工具执行结果转为 function_call_output，并绑定正确 call_id。
  • reasoning.effort 从 AGENT_REASONING_EFFORT 读取，默认 medium。
  • store 默认 false，避免教学 CLI 无意保存响应。
  • 失败响应返回包含 status code 和 body 的 error。

工具 schema 转换：

func toOpenAIFunctionTools(specs []ToolSpec) []map[string]any {
	tools := make([]map[string]any, 0, len(specs))
	for _, spec := range specs {
		tools = append(tools, map[string]any{
			"type":        "function",
			"name":        spec.Name,
			"description": spec.Description,
			"parameters":  spec.InputSchema,
		})
	}
	return tools
}

响应转换规则：

  • output[].type == "function_call" -> ContentBlock{Type:"tool_use", ID: call_id, Name: name, Input: parsed arguments}。
  • output[].type == "message" 中的文本 -> ContentBlock{Type:"text", Text: text}。
  • 只要存在 function call，统一返回 StopReason: "tool_use"。
  • reasoning items：保留在 provider transcript 中；当 OpenAI 返回需要回传的 reasoning item 时，下一轮 Responses input 必须包含上一轮 output item 和本地 function_call_output item。

Task 13: 实现 s12 worktree 任务隔离

Prerequisites: Task 8 (持久任务图)

Goal: 实现 git worktree 管理，任务与 worktree 双向绑定，支持创建、运行命令、保留和移除 worktree，操作事件写入 events.jsonl。

给第一次接触 Agent 的读者

源教程 s12 解决并行 agent 最实际的问题：多个任务不能都在同一个工作目录里改文件。任务图回答“做什么”，worktree 回答“在哪里做”。把两者绑定后，每个任务都能拥有自己的隔离目录和分支。

新手先懂这几个概念

  • Git worktree：同一个 Git 仓库可以签出多个工作目录，每个目录有自己的文件状态和分支。
  • 控制面：.tasks/ 记录任务状态、owner、绑定的 worktree 名称。
  • 执行面：.worktrees/ 下的隔离目录负责真实命令和代码修改。
  • 事件日志：events.jsonl 记录 create/remove/keep/failed，方便崩溃后理解发生过什么。

和源教程的对应关系

源 s12 把 task ID 和 worktree name 绑定。本任务用 internal/worktree/manager.go 封装 git worktree 命令，并在创建、执行、保留、移除时同步更新 task 状态和 worktree index。

完成后应该理解

你应该能说清为什么 worktree 是团队 agent 的安全阀：它让并行修改有边界，让完成、保留、删除都成为显式动作，而不是混在同一个脏工作区里。
